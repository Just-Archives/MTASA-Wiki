Server

هذا المثال يُحضر لاعب عشوائي ويضع عليه العلامة ذات الرقم 52 

-- اختيار لاعب عشوائي

function setupRandomRobber (  )
	local myPlayer = getRandomPlayer (  ); -- جلب لاعب عشوائي
	local myBlip = createBlipAttachedTo ( myPlayer, 52 ); -- نعمل إنشاء بلب يتبع اللاعب
end -- نغلق الـ function
addCommandHandler ( "createBlipAttachedTo", setupRandomRobber ); -- إنشاء الأمر

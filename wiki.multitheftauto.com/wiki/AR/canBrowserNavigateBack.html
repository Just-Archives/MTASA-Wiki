AR/canBrowserNavigateBack

.هذه الوظيفة تتحقق اذا كان المتصفح بامكانه العودة للصفحة السابقة ام لا

تركيب الوظيفة

bool canBrowserNavigateBack( browser webBrowser )

الفراغات المطلوبة

  • .المتصفح: المتصفح الذي تود التحقق منه إذا كان بامكانه العودة للصفحة السابقة

المعطيات

اذا كان المتصفح لا يستطيع العودة false اذا كان المتصفح يقبل العودة للصفحة السابقة و true تعود ب

مثال

Click to collapse [-]
مثال

هذا المثال يقوم بعمل متصفح يقبل بالعودة للصفحات الاخيرة والانتقال للتالية وانعاش الصفحة ايضا 

window = guiCreateWindow(126, 122, 848, 674, "browser", false)
guiWindowSetSizable(window, false)
navigateBackBtn = guiCreateButton(10, 19, 39, 36, "<", false, window)
navigateForwardBtn = guiCreateButton(98, 19, 39, 36, ">", false, window)
reloadBtn = guiCreateButton(49, 19, 49, 36, "reload", false, window)
addressBar =  guiCreateEdit(137, 19, 701, 36, "", false, window)
guiSetEnabled(addressBar,false)
webBrowser = guiCreateBrowser(10, 55, 828, 609, false, false, false, window)
local theBrowser = guiGetBrowser(webBrowser) 

-- اظهار الصفحة باستخدام الرابط الخاص بها وعرضها في المتصفح الذي قمنا بصنعه
addEventHandler("onClientBrowserCreated", theBrowser, 
    function()
	showCursor(true)
        loadBrowserURL(source, "https://forum.mtasa.com/")
    end
)

-- هنا نتحقق إذا كان المتصفح بامكانه التنقل في أي اتجاه وهل يقبل بالذهاب للصفحة التالية أو العودة للصفحة السابقة
addEventHandler( "onClientBrowserDocumentReady", theBrowser, function( url )
guiSetText( addressBar, getBrowserURL( theBrowser ) )
end )

-- هذا الجزء يفعل الأزرار علي العودة للصفحة السابقة والذهاب للصفحة التالية وانعاش الصفحة
addEventHandler( "onClientGUIClick", resourceRoot, function ( )
    if source == navigateBackBtn then -- هذا الجزء يفعل الأزرار علي العودة للصفحة السابقة والذهاب للصفحة التالية وانعاش الصفحة
	if canBrowserNavigateBack(theBrowser) ~= true then return end -- اذا كان المتصفح لا يتقبل العودة فيلغي الامر
        navigateBrowserBack(theBrowser) -- العودة للصفحة السابقة
    elseif source == navigateForwardBtn then -- اذا كان الزر المضغوط عليه هو زر الانتقال للصفحة التالية
	if canBrowserNavigateForward(theBrowser) ~= true then return end -- اذا كان المتصفح لا يقبل الذهاب للصفحة التالية فيلغي الامر
        navigateBrowserForward(theBrowser) -- الانتقال للصفحة التالية
    elseif source == reloadBtn then -- اذا كان الزر المضغوط عليه هو انعاش الصفحة
        reloadBrowserPage(theBrowser) -- نقوم بانعاش الصفحة الحالية
    end
end )
